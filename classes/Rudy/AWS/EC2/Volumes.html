Creates a Rudy::AWS::EC2::Volume object from:

    size: "1"
    snapshotId:
    availabilityZone: us-east-1b
    attachmentSet:
    createTime: "2009-03-17T20:10:48.000Z"
    volumeId: vol-48826421
    attachmentSet:
      item:
      - attachTime: "2009-03-17T21:49:54.000Z"
        status: attached
        device: /dev/sdh
        instanceId: i-956af3fc

133:       def list_as_hash(state=nil, vol_id=[], &each_vol)
134:         state &&= state.to_sym
135:         state = nil if state == :any
136:         # A nil state is fine, but we don't want an unknown one!
137:         raise UnknownState, state if state && !Volumes.known_state?(state)
138:       
139:         opts = { 
140:           :volume_id => vol_id ? [vol_id].flatten : [] 
141:         }
142: 
143:         vlist = Rudy::AWS::EC2.execute_request({}) { 
144:           @@ec2.describe_volumes(opts) 
145:         }
146: 
147:         volumes = {}
148:         return volumes unless vlist['volumeSet'].is_a?(Hash)
149:         vlist['volumeSet']['item'].each do |vol|
150:           v = Volumes.from_hash(vol)
151:           next if state && v.state != state.to_s
152:           volumes[v.awsid] = v
153:         end
154:         volumes.values.each { |v| each_vol.call(v) } if each_vol
155:         volumes = nil if volumes.empty?
156:         volumes
