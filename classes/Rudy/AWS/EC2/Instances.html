from_hash (h)

h is a hash of instance properties in the format returned by EC2::Base#describe_instances: 

kernelId: aki-9b00e5f2
amiLaunchIndex: "0"
keyName: solutious-default
launchTime: "2009-03-14T12:48:15.000Z"
instanceType: m1.small
imageId: ami-0734d36e
privateDnsName:
reason:
placement:
  availabilityZone: us-east-1b
dnsName:
instanceId: i-cdaa34a4
instanceState:
  name: pending
  code: "0"

Returns an Instance object.

console (inst_id, &each_inst)

System console output.

  • inst_id instance ID (String) or Instance object.

NOTE: Amazon sends the console outputs as a Base64 encoded string. This method DOES NOT decode in order to remain compliant with the data formats returned by Amazon.

You can decode it like this: 

require 'base64'
Base64.decode64(output)

create (opts={}, &each_inst)

Return an Array of Instance objects. Note: These objects will not have DNS data because they will still be in pending state. The DNS info becomes available once the instance enters the running state.

opts supports the following parameters:

  • :zone
  • :ami
  • :group
  • :size
  • :keypair
  • :private true or false (default)
  • :machine_data
  • :min count
  • :max count

191:       def list_as_hash(state=nil, inst_ids=[], &each_inst)
192:         state &&= state.to_sym
193:         state = nil if state == :any
194:         raise "Unknown state: #{state}" if state && !Instances.known_state?(state)
195:         state = :'shutting-down' if state == :shutting_down # EC2 uses a dash
196: 
197:         # If we got Instance objects, we want just the IDs.
198:         # This method always returns an Array.
199:         inst_ids = objects_to_instance_ids(inst_ids)
200:       
201:         response = Rudy::AWS::EC2.execute_request({}) {
202:           @@ec2.describe_instances(:instance_id => inst_ids)
203:         }
204:       
205:         # requestId: c16878ac-28e4-4859-9878-ef93af45789c
206:         # reservationSet: 
207:         #   item: 
208:         #   - reservationId: r-e493148d
209:         #     groupSet: 
210:         #       item: 
211:         #       - groupId: default
212:         #     instancesSet: 
213:         #       item:
214:         return nil unless response['reservationSet'].is_a?(Hash)  # No instances 
215:       
216:         resids = []
217:         instances = {}
218:         response['reservationSet']['item'].each do |res|      
219:           resids << res['reservationId']
220:           groups = res['groupSet']['item'].collect { |g| g['groupId'] }
221:           # And each reservation can have 1 or more instances
222:           next unless res['instancesSet'].is_a?(Hash)
223:           res['instancesSet']['item'].each do |props|
224:             inst = Instances.from_hash(props)
225:             next if state && inst.state != state.to_s
226:             inst.groups = groups
227:             #puts "STATE: #{inst.state} #{state}"
228:             instances[inst.awsid] = inst
229:           end
230:         end
231:         
232:         instances.each_value { |inst| each_inst.call(inst) } if each_inst
233:         
234:         instances = nil if instances.empty? # Don't return an empty hash
235:         instances
236:       end
